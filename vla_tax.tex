\newcommand{\ResolveTagColors}[1]{%
  \edef\ResolvedTagKey{#1}%
  % Default to E-style green when tag key is missing.
  \def\CurrentTagFill{taggreenfill}%
  \def\CurrentTagText{taggreentext}%
  \ifcsname tagFill@\ResolvedTagKey\endcsname
    \edef\CurrentTagFill{\TagFill{\ResolvedTagKey}}%
    \edef\CurrentTagText{\TagText{\ResolvedTagKey}}%
  \fi
}
\newcommand{\DrawSmallTagAt}[3]{%
  \edef\ResolvedTagToken{#3}%
  \ResolveTagColors{\ResolvedTagToken}%
  \node[anchor=north west, minimum width=34, minimum height=22,
        inner sep=0pt, rounded corners=5pt, fill=\CurrentTagFill, draw=none,
        font=\fontsize{11.5}{11.5}\selectfont\bfseries, text=\CurrentTagText]
    at (#1,#2) {+\ResolvedTagToken};
}

\newcommand{\DrawLegendTagAt}[3]{%
  \edef\ResolvedTagToken{#3}%
  \ResolveTagColors{\ResolvedTagToken}%
  \node[anchor=north west, minimum width=49, minimum height=30,
        inner sep=0pt, rounded corners=7pt, fill=\CurrentTagFill, draw=none,
        font=\fontsize{15}{15}\selectfont\bfseries, text=\CurrentTagText]
    at (#1,#2) {+\ResolvedTagToken};
}

\newcommand{\DrawCardTags}[5]{%
  \edef\ResolvedTagA{#4}%
  \edef\ResolvedTagB{#5}%
  \ifdefempty{\ResolvedTagA}{%
    % No tags for this card.
  }{%
  \pgfmathsetmacro{\tagy}{#2 + 4}
  \ifdefempty{\ResolvedTagB}{%
    \pgfmathsetmacro{\tagx}{#1 + #3 - 39}
    \DrawSmallTagAt{\tagx}{\tagy}{\ResolvedTagA}%
  }{%
    \pgfmathsetmacro{\tagxA}{#1 + #3 - 78}
    \pgfmathsetmacro{\tagxB}{#1 + #3 - 39}
    \DrawSmallTagAt{\tagxA}{\tagy}{\ResolvedTagA}%
    \DrawSmallTagAt{\tagxB}{\tagy}{\ResolvedTagB}%
  }%
  }%
}

\newcommand{\MilestoneCard}[1]{%
  \pgfkeys{/milestone/.cd,tag1=,tag2=,authors=,ref=,connector=on,titlecolor=,topshift=0,#1}%
  \pgfmathsetmacro{\axisx}{\msAxis}
  \pgfmathsetmacro{\cardy}{\LaneY{\msLane}}
  \pgfmathsetmacro{\cardx}{\axisx - \msOffset}
  \pgfmathsetmacro{\cardcx}{\cardx + \msWidth/2}
  \pgfmathsetmacro{\headery}{\cardy + \CardYearYPad + \msTopShift}
  \pgfmathsetmacro{\authory}{\cardy + \CardAuthorsYPad + \msTopShift}
  \pgfmathsetmacro{\yearx}{\cardcx - \CardYearTitleGapX}
  \pgfmathsetmacro{\titlex}{\cardcx + \CardYearTitleGapX}
  \def\msAuthorsExpanded{\msAuthors}%
  \def\msRefExpanded{\msRef}%
  \ifdefempty{\msAuthorsExpanded}{%
    \pgfmathsetmacro{\bodyy}{\headery + \CardBodyGapFromHeader}
  }{%
    \pgfmathsetmacro{\bodyy}{\headery + \CardBodyGapWithAuthors}
  }%
  \edef\msTagAExpanded{\msTagA}%
  \edef\msTagBExpanded{\msTagB}%
  \edef\msTitleColorExpanded{\msTitleColor}%
  \ifdefempty{\msTitleColorExpanded}{%
    \def\ResolvedTitleColor{\CatTitle{\msCategory}}%
  }{%
    \def\ResolvedTitleColor{\msTitleColorExpanded}%
  }%
  \draw[draw=\CatBorder{\msCategory}, fill=\CatFill{\msCategory},
        line width=1.05pt, rounded corners=11pt]
    (\cardx,\cardy) rectangle ++(\msWidth,\msHeight);
  \DrawCardTags{\cardx}{\cardy}{\msWidth}{\msTagAExpanded}{\msTagBExpanded}

  \node[anchor=north west, align=left, font=\fontsize{13}{15}\selectfont, text=black]
    at (\cardx+\CardYearXPad,\headery) {\msYear};

  \ifdefempty{\msRefExpanded}{%
    \node[anchor=north west, align=left, text=\ResolvedTitleColor,
        font=\fontsize{\msTitleSize}{\msTitleLeading}\selectfont\bfseries\boldmath]
      at (\cardx+\CardTitleXPad,\headery) {\msTitle};
  }{%
    \node[anchor=north west, align=left, text=\ResolvedTitleColor,
        font=\fontsize{\msTitleSize}{\msTitleLeading}\selectfont\bfseries\boldmath]
      at (\cardx+\CardTitleXPad,\headery) {\msTitle\mbox{\textnormal{~\cite{\msRefExpanded}}}};
  }%

  \ifdefempty{\msAuthorsExpanded}{%
  }{%
    \node[anchor=north west, align=left, text=graytext,
          font=\fontsize{12.15}{14.75}\selectfont]
      at (\cardx+\CardAuthorsXPad,\authory) {\msAuthorsExpanded};
  }%

  \node[anchor=north, align=center, text=black,
        font=\fontsize{12.15}{14.75}\selectfont]
    at (\cardcx,\bodyy) {\msBody};

  % Dynamic connector rule:
  % card above timeline -> connector starts at card bottom edge
  % card below timeline -> connector starts at card top edge
  \edef\msConnectorExpanded{\msConnector}%
  \ifnum\pdfstrcmp{\msConnectorExpanded}{off}=0\relax
  \else
    \pgfmathsetmacro{\connectorStartY}{\cardy < \TimelineY ? \cardy + \msHeight : \cardy}
    \pgfmathsetmacro{\connectorJoinY}{\cardy < \TimelineY ? \JoinAboveY : \JoinBelowY}
    \draw[dash pattern=on 6pt off 4pt, line width=1.0pt, draw=dashblue]
      (\axisx,\connectorStartY) -- (\axisx,\connectorJoinY);
  \fi
}
